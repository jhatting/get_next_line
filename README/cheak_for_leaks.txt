2. insert the while (1); at the end of the function before the last parensisthes like this :
#include "get_next_line.h"
#include <fcntl.h>
int        main(int argc, char **argv)
{
    int        fd;
    char    *line;
    if (argc == 1)
        fd = 0;
    else if (argc == 2)
        fd = open(argv[1], O_RDONLY);
    else
        return (2);
    while (get_next_line(fd, &line) == 1)
    {
        ft_putendl(line);
        free(line);
    }
    if (argc == 2)
        close(fd);
		while (1);
}
